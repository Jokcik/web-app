# Первый запуск
npm install -g @angular/cli
npm install -g typescript
npm install -g ts-node
npm install -g nodemon

# Запуск сервеса с помощью nodemon
npm start

# Скопилировать стили
npm run css
npm run csswatch

# компиляция сервера на прод
npm run prod

# Добавить изменения на сервер
git push heroku master
